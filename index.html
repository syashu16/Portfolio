import numpy as np
from sklearn.linear_model import LinearRegression

# Sample data

# Create and train model
model = LinearRegression()
model.fit(X, y)

# Make prediction
prediction = model.predict([[6]])
print(f"Prediction for x=6: {prediction[0]:.2f}")
